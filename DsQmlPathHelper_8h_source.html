83 Q_INVOKABLE QString arrow(float x0, float y0, float x1, float y1, float thickness, float width = 4,
84 float length = 4, float concavity = 0) {
85 // Convert input points to QPointF
86 QPointF p0(x0, y0);
87 QPointF p1(x1, y1);
88
89 // Calculate distance and direction
90 float distance = qSqrt(qPow(p1.x() - p0.x(), 2) + qPow(p1.y() - p0.y(), 2));
91 if (distance < 1e-6) return QString("M %1 %2 Z").arg(x0).arg(y0); // Degenerate case
92
93 QPointF direction((p1.x() - p0.x()) / distance, (p1.y() - p0.y()) / distance);
94
95 // Calculate normal vector
96 QPointF normal(0.5f * thickness * direction.y(), -0.5f * thickness * direction.x());
97
98 // Calculate base point
99 QPointF base = p0 + direction * qMax(0.0f, distance - thickness * length);
100
101 // Precompute points
102 QPointF pt1 = p0 - normal; // Left side of shaft
103 QPointF concave_offset = direction * (thickness * length * concavity);
104 QPointF pt2 = base - normal + concave_offset; // Left concave point
105 QPointF width_offset = normal * width;
106 QPointF pt3 = base - width_offset; // Left base of arrowhead
107 QPointF pt4 = p1; // Arrowhead tip
108 QPointF pt5 = base + width_offset; // Right base of arrowhead
109 QPointF pt6 = base + normal + concave_offset; // Right concave point
110 QPointF pt7 = p0 + normal; // Right side of shaft
111
112 // Build SVG path
113 return QString("M %1 %2 L %3 %4 L %5 %6 L %7 %8 L %9 %10 L %11 %12 L %13 %14 Z")
114 .arg(pt1.x())
115 .arg(pt1.y()) // Point 1
116 .arg(pt2.x())
117 .arg(pt2.y()) // Point 2
118 .arg(pt3.x())
119 .arg(pt3.y()) // Point 3
120 .arg(pt4.x())
121 .arg(pt4.y()) // Point 4
122 .arg(pt5.x())
123 .arg(pt5.y()) // Point 5
124 .arg(pt6.x())
125 .arg(pt6.y()) // Point 6
