41 template <typename Traits, typename OutputPtr, typename InputPtr> inline
42 int toUtf8(ushort u, OutputPtr &dst, InputPtr &src, InputPtr end)
43 {
44 if (!Traits::skipAsciiHandling && u < 0x80)
45 {
46 // U+0000 to U+007F (US-ASCII) - one byte
47 Traits::appendByte(dst, uchar(u));
48 return 0;
49 }
50 else if (u < 0x0800)
51 {
52 // U+0080 to U+07FF - two bytes
53 // first of two bytes
54 Traits::appendByte(dst, 0xc0 | uchar(u >> 6));
55 }
56 else
57 {
58 if (!QChar::isSurrogate(u))
59 {
60 // U+0800 to U+FFFF (except U+D800-U+DFFF) - three bytes
61 if (!Traits::allowNonCharacters && QChar::isNonCharacter(u))
62 return Traits::Error;
63 // first of three bytes
64 Traits::appendByte(dst, 0xe0 | uchar(u >> 12));
65 }
66 else
67 {
68 // U+10000 to U+10FFFF - four bytes
69 // need to get one extra codepoint
70 if (Traits::availableUtf16(src, end) == 0)
71 return Traits::EndOfString;
72 ushort low = Traits::peekUtf16(src);
73 if (!QChar::isHighSurrogate(u))
74 return Traits::Error;
75 if (!QChar::isLowSurrogate(low))
76 return Traits::Error;
77 Traits::advanceUtf16(src);
78 uint ucs4 = QChar::surrogateToUcs4(u, low);
79 if (!Traits::allowNonCharacters && QChar::isNonCharacter(ucs4))
80 return Traits::Error;
81 // first byte
82 Traits::appendByte(dst, 0xf0 | (uchar(ucs4 >> 18) & 0xf));
83 // second of four bytes
84 Traits::appendByte(dst, 0x80 | (uchar(ucs4 >> 12) & 0x3f));
85 // for the rest of the bytes
86 u = ushort(ucs4);
87 }
88 // second to last byte
89 Traits::appendByte(dst, 0x80 | (uchar(u >> 6) & 0x3f));
90 }
91 // last byte
92 Traits::appendByte(dst, 0x80 | (u & 0x3f));
93 return 0;
94 }
95 inline bool isContinuationByte(uchar b)
96 {
97 return (b & 0xc0) == 0x80;
98 }
101 template <typename Traits, typename OutputPtr, typename InputPtr> inline
102 int fromUtf8(uchar b, OutputPtr &dst, InputPtr &src, InputPtr end)
103 {
104 int charsNeeded;
105 uint min_uc;
106 uint uc;
107 if (!Traits::skipAsciiHandling && b < 0x80)
108 {
109 // US-ASCII
110 Traits::appendUtf16(dst, b);
111 return 1;
112 }
113 if (!Traits::isTrusted && Q_UNLIKELY(b <= 0xC1))
114 {
115 // an UTF-8 first character must be at least 0xC0
116 // however, all 0xC0 and 0xC1 first bytes can only produce overlong sequences
117 return Traits::Error;
118 }
119 else if (b < 0xe0)
120 {
121 charsNeeded = 2;
122 min_uc = 0x80;
123 uc = b & 0x1f;
124 }
125 else if (b < 0xf0)
126 {
127 charsNeeded = 3;
128 min_uc = 0x800;
129 uc = b & 0x0f;
130 }
131 else if (b < 0xf5)
132 {
133 charsNeeded = 4;
134 min_uc = 0x10000;
135 uc = b & 0x07;
136 }
137 else
138 {
139 // the last Unicode character is U+10FFFF
140 // it's encoded in UTF-8 as "\xF4\x8F\xBF\xBF"
141 // therefore, a byte higher than 0xF4 is not the UTF-8 first byte
142 return Traits::Error;
143 }
144 int bytesAvailable = Traits::availableBytes(src, end);
145 if (Q_UNLIKELY(bytesAvailable < charsNeeded - 1))
146 {
147 // it's possible that we have an error instead of just unfinished bytes
148 if (bytesAvailable > 0 && !isContinuationByte(Traits::peekByte(src, 0)))
149 return Traits::Error;
150 if (bytesAvailable > 1 && !isContinuationByte(Traits::peekByte(src, 1)))
151 return Traits::Error;
152 return Traits::EndOfString;
153 }
154 // first continuation character
155 b = Traits::peekByte(src, 0);
156 if (!isContinuationByte(b))
157 return Traits::Error;
158 uc <<= 6;
159 uc |= b & 0x3f;
160 if (charsNeeded > 2)
161 {
162 // second continuation character
163 b = Traits::peekByte(src, 1);
164 if (!isContinuationByte(b))
165 return Traits::Error;
166 uc <<= 6;
167 uc |= b & 0x3f;
168 if (charsNeeded > 3)
169 {
170 // third continuation character
171 b = Traits::peekByte(src, 2);
172 if (!isContinuationByte(b))
173 return Traits::Error;
174 uc <<= 6;
175 uc |= b & 0x3f;
176 }
177 }
178 // we've decoded something; safety-check it
179 if (!Traits::isTrusted)
180 {
181 if (uc < min_uc)
182 return Traits::Error;
183 if (QChar::isSurrogate(uc) || uc > QChar::LastValidCodePoint)
184 return Traits::Error;
185 if (!Traits::allowNonCharacters && QChar::isNonCharacter(uc))
186 return Traits::Error;
187 }
188 // write the UTF-16 sequence
189 if (!QChar::requiresSurrogates(uc))
190 {
191 // UTF-8 decoded and no surrogates are required
192 // detach if necessary
193 Traits::appendUtf16(dst, ushort(uc));
194 }
195 else
196 {
197 // UTF-8 decoded to something that requires a surrogate pair
198 Traits::appendUcs4(dst, uc);
199 }
200 Traits::advanceByte(src, charsNeeded - 1);
201 return charsNeeded;
202 }

205struct QUtf8BaseTraits

207 static const bool isTrusted = false;
208 static const bool allowNonCharacters = true;
209 static const bool skipAsciiHandling = false;
210 static const int Error = -1;
211 static const int EndOfString = -2;
212 static bool isValidCharacter(uint u)
213 {
214 return int(u) >= 0;
215 }
216 static void appendByte(uchar *&ptr, uchar b)
217 {
218 *ptr++ = b;
219 }
220 static uchar peekByte(const uchar *ptr, int n = 0)
221 {
222 return ptr[n];
223 }
224 static qptrdiff availableBytes(const uchar *ptr, const uchar *end)
225 {
226 return end - ptr;
227 }
228 static void advanceByte(const uchar *&ptr, int n = 1)
229 {
230 ptr += n;
231 }
232 static void appendUtf16(ushort *&ptr, ushort uc)
233 {
234 *ptr++ = uc;
235 }
236 static void appendUcs4(ushort *&ptr, uint uc)
237 {
238 appendUtf16(ptr, QChar::highSurrogate(uc));
239 appendUtf16(ptr, QChar::lowSurrogate(uc));
240 }
241 static ushort peekUtf16(const ushort *ptr, int n = 0)
242 {
243 return ptr[n];
244 }
245 static qptrdiff availableUtf16(const ushort *ptr, const ushort *end)
246 {
247 return end - ptr;
248 }
249 static void advanceUtf16(const ushort *&ptr, int n = 1)
250 {
251 ptr += n;
252 }
253 // it's possible to output to UCS-4 too
254 static void appendUtf16(uint *&ptr, ushort uc)
255 {
256 *ptr++ = uc;
257 }
258 static void appendUcs4(uint *&ptr, uint uc)
259 {
260 *ptr++ = uc;
261 }
